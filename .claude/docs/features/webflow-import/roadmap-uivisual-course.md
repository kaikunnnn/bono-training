## 13. データ構造（実装用）

### コース情報
```typescript
interface RoadmapCourse {
  slug: string;              // "uivisual-course"
  title: string;             // "UIビジュアル入門コース"
  description: string;       // コース説明文
  estimatedDuration: string; // "1~2ヶ月"
  price: string;             // "￥5,800~/月"
  steps: RoadmapStep[];
}
```

### ステップ情報
```typescript
interface RoadmapStep {
  number: string;            // "01", "02", etc.
  title: string;             // ステップタイトル
  goal: string;              // ゴールテキスト
  estimatedDuration: string; // "3〜7日", "1〜2週間", etc.
  description: string;       // 説明文
  contents: RoadmapContent[];
}
```

### コンテンツ情報
```typescript
interface RoadmapContent {
  label: string;             // "チュートリアル", "実践して理解", etc.
  title: string;             // コンテンツタイトル
  description: string;       // 説明文
  thumbnail?: string;        // サムネイル画像URL
  link: string;              // 詳細ページへのリンク
  note?: string;             // 備考（ベータ版など）
}
```

### 関連コース情報
```typescript
interface RelatedCourse {
  title: string;             // コースタイトル
  description: string;       // 説明文
  link: string;              // リンクパス
  tags: string[];            // タグ配列
}
```
